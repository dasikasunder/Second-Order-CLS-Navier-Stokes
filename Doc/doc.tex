	\begin{abstract}
		In this paper, we discuss the use of face based constrained least-squares reconstruction for a second-order cell-centered finite volume discretization of incompressible flows on unstructured grids. In a cell-centered finite volume method, although variables at cell centers are updated, calculation of fluxes and application of boundary conditions is performed at faces of the mesh. 
	\end{abstract}
	
	\end{frontmatter}

	\subsection{Constrained Least Squares}
	The constrained least squares is \cite{boyd2018introduction} used for reconstruction \cite{gradient-calculation-methods} to solve incompressible Navier-Stokes \cite{KIM2000411}
	
	\subsection{Reconstruction for interior faces}
	\noindent The constrained least-squares system for reconstruction on an interior face of the mesh is
	\begin{equation}
		\begin{bmatrix}
			2(N_c-2) & 2\sum \Delta x_i & 2\sum \Delta y_i & 1 & 1\\
			2\sum \Delta x_i & 2\sum \Delta x_i^2 & 2\sum \Delta x_i \Delta y_i & \Delta x_1 & \Delta x_2 \\
			2\sum \Delta y_i & 2\sum \Delta x_i \Delta y_i & 2\sum \Delta y_i^2 & \Delta y_1 & \Delta y_2 \\
			1 & \Delta x_1 & \Delta y_1 & 0 & 0 \\
			1 & \Delta x_2 & \Delta y_2 & 0 & 0 \\
		\end{bmatrix}
		\begin{bmatrix}
			\phi_0 \\
			\phi_x \\
			\phi_y \\
			\hat{z}_1 \\ 
			\hat{z}_2
		\end{bmatrix}	
		= 
		\begin{bmatrix}
			2\sum \phi_i \\
			2\sum \Delta x_i \phi_i  \\ 
			2\sum \Delta y_i \phi_i \\
			\phi_1 \\
			\phi_2 \\ 
		\end{bmatrix}	
	\end{equation}  
	
	\subsection{Reconstruction for faces on Dirichlet boundaries}
	
	\begin{equation}
		\begin{bmatrix}
			2(N_c-2) & 2\sum \Delta x_i & 2\sum \Delta y_i & 1 & 1\\
			2\sum \Delta x_i & 2\sum \Delta x_i^2 & 2\sum \Delta x_i \Delta y_i & \Delta x_1 & 0 \\
			2\sum \Delta y_i & 2\sum \Delta x_i \Delta y_i & 2\sum \Delta y_i^2 & \Delta y_1 & 0 \\
			1 & \Delta x_1 & \Delta y_1 & 0 & 0 \\
			1 & 0 & 0 & 0 & 0 \\
		\end{bmatrix}
		\begin{bmatrix}
			\phi_0 \\
			\phi_x \\
			\phi_y \\
			\hat{z}_1 \\ 
			\hat{z}_2
		\end{bmatrix}	
		= 
		\begin{bmatrix}
			2\sum \phi_i \\
			2\sum \Delta x_i \phi_i  \\ 
			2\sum \Delta y_i \phi_i \\
			\phi_1 \\
			\phi_b \\ 
		\end{bmatrix}	
	\end{equation} 
	
	\subsection{Reconstruction for faces on Neumann boundaries}
	
	\begin{equation}
		\begin{bmatrix}
			2(N_c-2) & 2\sum \Delta x_i & 2\sum \Delta y_i & 1 & 0\\
			2\sum \Delta x_i & 2\sum \Delta x_i^2 & 2\sum \Delta x_i \Delta y_i & \Delta x_1 & n_x \\
			2\sum \Delta y_i & 2\sum \Delta x_i \Delta y_i & 2\sum \Delta y_i^2 & \Delta y_1 & n_y \\
			1 & \Delta x_1 & \Delta y_1 & 0 & 0 \\
			0 & n_x & n_y & 0 & 0 \\
		\end{bmatrix}
		\begin{bmatrix}
			\phi_0 \\
			\phi_x \\

	\subsection{Potential flow around a circular cylinder}
	\noindent Potential flow over a cylinder models inviscid, incompressible, and irrotational fluid flowing around a circular body. Since the velocity field is irrotational, it can be expressed in terms of a potential function $\phi$ \cite{thomas2015thomas}, as  $\mathbf{u} = \nabla \phi$. Further, the divergence free condition on velocity, $\nabla \cdot \mathbf{u}$, leads to a Poisson equation, $\nabla^2 \phi = 0$, for the potential function. For potential flow past cylinder, the exact solution for radial and tangential velocity components $u_r$ and $u_{\theta}$, is given in polar coordinates $(r,\theta)$ by
	\begin{subequations}
		\begin{equation}
		\phi = \left(r + \frac{R^2}{r}\right)U\cos \theta
		\end{equation}
		\begin{equation}
			u_r  = \frac{\partial \phi}{\partial r} =  \left(1 - \frac{R^2}{r^2}\right)U\cos \theta
		\end{equation}
		\begin{equation}
			u_{\theta}  = \frac{1}{r}\frac{\partial \phi}{\partial \theta} =  -\left(1 + \frac{R^2}{r^2}\right)U\sin \theta
		\end{equation}
		\label{eq:potential_flow}
	\end{subequations}
  	For our test case, we use free stream velocity $U$ = 1. The computational domain is a square $\Omega = [-5,5]^2$ at the center of which  a circular cylinder of radius $R = 1$ is placed. On surface of the cylinder, no flow through condition, $\mathbf{u} \cdot \mathbf{n} = 0$, translates to a homogeneous Neumann condition, $\nabla \phi \cdot \mathbf{n} = 0$, for the potential function. On boundaries of the outer square domain, we impose exact solution provided by Equation \ref{eq:potential_flow}. A schematic diagram for the problem showing the geometry and boundary conditions is shown in Figure \ref{fig:potential-flow-schematic}.

	\subsection{Kovasznay flow solution}
	\noindent The first test case for incompressible Navier-Stokes equations is the Kovasznay analytical solution for laminar flow \cite{Kovasznay_1948}. This test case provides a non-trivial exact solution for steady-state Navier-Stokes equations, and is given by 
	\begin{subequations}
		\begin{equation}
			u = 1 - e^{\lambda x} \cos (2\pi y)
		\end{equation}
		\begin{equation}
			v = \frac{\lambda}{2\pi} e^{\lambda x} \sin(2\pi y)
		\end{equation}
		\begin{equation}
			p = \frac{1}{2}(1 - e^{2\lambda x})
		\end{equation}
		\label{eq:kovasznay_flow}
	\end{subequations}
	In the above equations $\lambda = \frac{1}{2\nu} - \sqrt{\frac{1}{4\nu^2} + 4\pi^2}$.
